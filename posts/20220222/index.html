#空座自动抢座

from selenium import webdriver
from selenium.webdriver.chrome.options import Options
import time
import urllib.request
from urllib.request import URLError

option=webdriver.ChromeOptions()
#option.add_argument('headless')
driver = webdriver.Chrome(chrome_options=option)
error=[]




url=r'http://seat.lib.sdu.edu.cn/web/seat3?area=5&segment=2596026&day=2021-8-7&startTime=08:00&endTime=22:30' #投稿系统网址
driver.get(url)



time.sleep(2)
print("当前时间 "+str(time.strftime("%H%M", time.localtime()))+" 开始抢座")

while True:
    button1 = driver.find_elements_by_xpath("//li[@class='seat ava-icon']")
    if str(button1) == "[]":
        time.sleep(3)
        
        continue
    else:
        print('已有空座'+str(button1[0].get_attribute('data-no'))+'，即将开始抢座')
        button1[0].click()
        time.sleep(1)
        print('正在登陆')
        username = driver.find_element_by_id('un')
        username.send_keys(r'**********')

        password= driver.find_element_by_id('pd')
        password.send_keys(r'**********')

        login_button = driver.find_element_by_id('index_login_btn')
        login_button.click()#点击登陆
        driver.implicitly_wait(3) # 等待时间
        print('登陆成功')
        button1 = driver.find_elements_by_xpath("//li[@class='seat ava-icon']")
        button1[0].click()
        driver.implicitly_wait(3)
        button2 = driver.find_element_by_xpath('/html/body/div[11]/div/table/tbody/tr[3]/td/div[2]/button[2]')
        button2.click()
        paper = driver.find_element_by_xpath('/html/body/div[11]/div/table/tbody/tr[2]/td/div/div')
        print(paper.text)
        
        break
